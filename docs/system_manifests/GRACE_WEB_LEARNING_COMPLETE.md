## ğŸ—„ï¸ Database Schema

### KnowledgeSource Table
```sql
CREATE TABLE knowledge_sources (
    id INTEGER PRIMARY KEY,
    source_id VARCHAR(64) UNIQUE NOT NULL,
    source_type VARCHAR(32) NOT NULL,  -- web, github, forum, api
    url TEXT NOT NULL,
    domain VARCHAR(255) NOT NULL,
    title TEXT,
    scraped_at TIMESTAMP,
    
    -- Governance verification
    governance_approved BOOLEAN,
    hunter_verified BOOLEAN,
    constitutional_approved BOOLEAN,
    
    -- Content metadata
    word_count INTEGER,
    code_snippet_count INTEGER,
    content_hash VARCHAR(64) UNIQUE,
    trust_score FLOAT,
    
    -- Immutable logging
    immutable_log_hash VARCHAR(64),
    previous_hash VARCHAR(64),
    
    -- Storage
    storage_path TEXT,
    metadata JSON,
    created_at TIMESTAMP
);
```

### KnowledgeApplication Table
```sql
CREATE TABLE knowledge_applications (
    id INTEGER PRIMARY KEY,
    application_id VARCHAR(64) UNIQUE NOT NULL,
    source_id VARCHAR(64) REFERENCES knowledge_sources(source_id),
    
    -- Application details
    application_type VARCHAR(64),
    context TEXT,
    code_generated TEXT,
    
    -- Sandbox testing
    tested_in_sandbox BOOLEAN,
    sandbox_passed BOOLEAN,
    sandbox_results JSON,
    
    -- Verification
    kpi_met BOOLEAN,
    trust_threshold_met BOOLEAN,
    governance_approved BOOLEAN,
    
    -- Outcomes
    success BOOLEAN,
    outcome_description TEXT,
    immutable_log_hash VARCHAR(64),
    applied_at TIMESTAMP
);
```
