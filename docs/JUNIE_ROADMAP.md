## Code Standards for Junie

### TypeScript/React Standards
```tsx
// Always include types
interface Props {
  taskId: number;
  onApprove: () => void;
}

// Use hooks for state
const [data, setData] = useState<Data | null>(null);

// Error boundaries
try {
  const result = await api.getData();
} catch (error) {
  showError(error.message);
}

// Loading states
if (loading) return <Loading />;
if (error) return <Error message={error} />;

// Accessible
<button aria-label="Approve task" onClick={...}>
```

### Python/FastAPI Standards
```python
# Type hints always
async def get_data(id: int) -> Dict[str, Any]:
    pass

# Error handling
try:
    result = await operation()
except Exception as e:
    logger.error(f"Failed: {e}")
    raise HTTPException(500, str(e))

# Metrics publishing
from backend.metric_publishers import DomainMetrics
await DomainMetrics.publish_something(value)

# Response models
from pydantic import BaseModel

class TaskResponse(BaseModel):
    task_id: int
    status: str
    result: Dict[str, Any]
```
